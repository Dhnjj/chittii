#include<stdio.h>
int main()
{
 int pno[10],bt[10],wt[10],tat[10],tp,twt=0,ttat=0,delay=0,pr[10];
 int i,j,temp;
 float awt,atat;
 printf("Priority Scheduling\n\n");
 printf("\nEnter the No of processes:");
 scanf("%d",&tp);
 for(i=0;i<tp;i++)
 {
 printf("Enter process no");
 scanf("%d",&pno[i]);
 printf("Enter the burst time for process %d:",pno[i]);
 scanf("%d",&bt[i]);
 printf("Enter the Priority for process %d:",pno[i]);
 scanf("%d",&pr[i]);
 wt[i]=0;
 }
 for(i=0;i<tp;i++)
 {
 for(j=i;j<tp;j++)
 {
 if(pr[i]>pr[j])
 {
 temp=pno[i];
 pno[i]=pno[j];
 pno[j]=temp;
 temp=bt[i];
 bt[i]=bt[j];
 bt[j]=temp;
 temp=pr[i];
 pr[i]=pr[j];
 pr[j]=temp;
 }
 }
 }
 wt[0]=0;
 tat[0]=wt[0]+bt[0];
 for(i=1;i<tp;i++)
 {
 for(j=0;j<i;j++)
 {
 wt[i]=wt[i]+bt[j];
 }
 twt = twt+wt[i];
 tat[i]=wt[i]+bt[i];
 ttat=ttat+tat[i];
 }
 awt=(float)twt/tp;
 atat=(float)ttat/tp;
 printf("\nPNO\tBT\tPriority\tWT\tTAT\n");
 for(int i=0;i<tp;i++)
 {
 printf("%d\t%d\t%d\t\t\t%d\t%d\n",pno[i],bt[i],pr[i],wt[i],tat[i]);
 }
 printf("\nAverage Waiting Time:%f",awt);
 printf("\nAverage Turnaround time:%f",atat);
 return 0;
}
